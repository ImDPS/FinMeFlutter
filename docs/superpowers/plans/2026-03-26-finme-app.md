# FinMe App — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build FinMe — a glassmorphism-first personal finance app for Android + iOS that tracks expenses, budgets, and net worth for a single Indian user using Account Aggregator + manual entry.

**Architecture:** Feature-first Flutter project with Riverpod state management. All financial data lives in a local SQLCipher-encrypted Drift database (offline-first reads/writes), synced to Firebase Firestore (asia-south1, Mumbai) when online. Account Aggregator data flows through a Firebase Cloud Function that decrypts and normalises Setu FIU payloads before writing to Firestore.

**Tech Stack:** Flutter 3.x · Dart · Riverpod · Drift + SQLCipher · Firebase Auth (phone OTP) · Firestore (asia-south1) · Setu FIU AA SDK · local_auth · flutter_secure_storage · fl_chart · Lucide Icons · Dio

**Spec:** `docs/superpowers/specs/2026-03-26-personal-finance-app-design.md`

import 'package:flutter/material.dart';
import '../theme/app_colors.dart';

class SkeletonLoader extends StatefulWidget {
  final double width;
  final double height;
  final double borderRadius;

  const SkeletonLoader({
    super.key,
    required this.width,
    required this.height,
    this.borderRadius = 8,
  });

  @override
  State<SkeletonLoader> createState() => _SkeletonLoaderState();
}

class _SkeletonLoaderState extends State<SkeletonLoader>
    with SingleTickerProviderStateMixin {
  late AnimationController _controller;
  late Animation<double> _animation;

  @override
  void initState() {
    super.initState();
    _controller = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 1200),
    )..repeat(reverse: true);
    _animation = Tween<double>(begin: 0.3, end: 0.7).animate(_controller);
  }

  @override
  void dispose() {
    _controller.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return AnimatedBuilder(
      animation: _animation,
      builder: (_, __) => Container(
        width: widget.width,
        height: widget.height,
        decoration: BoxDecoration(
          color: AppColors.surface.withOpacity(_animation.value),
          borderRadius: BorderRadius.circular(widget.borderRadius),
        ),
      ),
    );
  }
}

import 'package:intl/intl.dart';

class CurrencyFormatter {
  CurrencyFormatter._();

  static final NumberFormat _inrFormat = NumberFormat.currency(
    locale: 'en_IN',
    symbol: '₹',
    decimalDigits: 0,
  );

  /// Formats an integer amount (paise or rupees) using Indian number system.
  /// e.g. 123456 → ₹1,23,456

    final secure = const FlutterSecureStorage();

    import 'dart:math';
    final rng = Random.secure();
    final bytes = List<int>.generate(32, (_) => rng.nextInt(256));
    return bytes.map((b) => b.toRadixString(16).padLeft(2, '0')).join();

import 'dart:async';
import 'dart:convert';
import 'package:connectivity_plus/connectivity_plus.dart';
import '../local/database.dart';
import '../remote/firestore_service.dart';

class SyncEngine {
  final AppDatabase db;
  final FirestoreService firestore;

  static const int _maxRetries = 3;
  static const List<int> _backoffSeconds = [2, 4, 8];

  SyncEngine({required this.db, required this.firestore});

  Future<void> flushQueue() async {
    final connectivity = await Connectivity().checkConnectivity();
    if (connectivity == ConnectivityResult.none) return;

    final pending = await db.syncQueueDao.getPendingItems();
    for (final item in pending) {
      await _processItem(item);
    }
  }

  Future<void> _processItem(SyncQueueItem item) async {
    if (item.retryCount >= _maxRetries) {
      await db.syncQueueDao.markFailed(item.id);
      return;
    }
    try {
      final payload = json.decode(item.payload) as Map<String, dynamic>;
      if (item.operation == 'upsert') {
        await firestore.upsert(payload['collection'], payload['id'], payload['data']);
      } else if (item.operation == 'delete') {
        await firestore.delete(payload['collection'], payload['id']);
      }
      await db.syncQueueDao.deleteItem(item.id);
    } catch (_) {
      final backoff = _backoffSeconds[item.retryCount.clamp(0, 2)];
      await Future.delayed(Duration(seconds: backoff));
      await db.syncQueueDao.incrementRetry(item.id);
    }
  }

  // Enqueue a write operation for offline-first syncing
  Future<void> enqueue({
    required String operation,
    required String collection,
    required String id,
    Map<String, dynamic>? data,
  }) async {
    final payload = json.encode({'collection': collection, 'id': id, 'data': data});
    await db.syncQueueDao.enqueue(
      SyncQueueCompanion.insert(
        id: Value(DateTime.now().microsecondsSinceEpoch.toString()),
        operation: operation,
        payload: payload,
        createdAt: DateTime.now(),
      ),
    );
  }
}

  }

  Future<void> setPin(String pin) async {
    // Simple SHA-256 hash (bcrypt not available in Dart without native bridge — use sha256 + salt)
    final hash = _hashPin(pin);
    await storage.write(key: _pinHashKey, value: hash);
    await storage.write(key: _failCountKey, value: '0');
  }

  Future<bool> verifyPin(String pin) async {
    // Check lockout
    final lockUntil = await storage.read(key: _lockUntilKey);
    if (lockUntil != null) {
      final lockTime = DateTime.parse(lockUntil);
      if (DateTime.now().isBefore(lockTime)) return false;
    }

    final stored = await storage.read(key: _pinHashKey);
    if (stored == null) return false;

    final correct = stored == _hashPin(pin);
    if (!correct) await _recordFailure();
    if (correct) await storage.write(key: _failCountKey, value: '0');
    return correct;
  }

  Future<void> _recordFailure() async {
    final countStr = await storage.read(key: _failCountKey) ?? '0';
    final count = int.parse(countStr) + 1;
    await storage.write(key: _failCountKey, value: count.toString());

    if (count >= 5) {
      // Lockout: 30s * 2^(count-5) — doubles each round
      final lockSeconds = 30 * (1 << (count - 5));
      final until = DateTime.now().add(Duration(seconds: lockSeconds));
      await storage.write(key: _lockUntilKey, value: until.toIso8601String());
    }
  }

  // crypto package already in pubspec from Task 1
  String _hashPin(String pin) {
    import 'dart:convert';
    import 'package:crypto/crypto.dart';
    final saltedPin = 'finme_salt_$pin';
    return sha256.convert(utf8.encode(saltedPin)).toString();
  }

  Future<void> signOut() async {
    await auth.signOut();
  }
}

  final String _baseUrl;
  final String _clientId;

    required String baseUrl,
    required String clientId,

  /// Initiates a consent request and returns the redirect URL for the AA webview.

    required String userId,
    required List<String> fiTypes, // e.g. ['DEPOSIT', 'MUTUAL_FUNDS']
  }) async {
    final response = await _dio.post(
      '$_baseUrl/consents',

        'x-client-secret': _clientSecret,
        'Content-Type': 'application/json',

      },
    );

  }
}

import * as functions from 'firebase-functions/v2/https';
import * as admin from 'firebase-admin';

admin.initializeApp();
const db = admin.firestore();

// Called by Setu after user approves consent.
// Receives encrypted FI data, decrypts, normalises, writes to Firestore.
export const receiveAAData = functions.onRequest(async (req, res) => {
  if (req.method !== 'POST') {
    res.status(405).send('Method not allowed');
    return;
  }

  const { userId, consentId, fiData, consentExpiryDate } = req.body;

  if (!userId || !fiData) {
    res.status(400).send('Missing required fields');
    return;
  }

  const batch = db.batch();
  const userRef = db.collection('users').doc(userId);

  // Normalise and write each account + transactions
  for (const account of fiData.accounts || []) {
    const accountRef = userRef.collection('accounts').doc(account.id);
    batch.set(accountRef, {
      id: account.id,
      name: account.maskedAccountNumber,
      type: account.accountType.toLowerCase(),
      institution: account.fipId,
      balance: Math.round(account.summary?.currentBalance || 0),
      lastSynced: admin.firestore.FieldValue.serverTimestamp(),
      consentExpiryDate: consentExpiryDate,
      consentStatus: 'active',
    }, { merge: true });

    for (const tx of account.transactions || []) {
      const txRef = userRef.collection('transactions').doc(`${account.id}_${tx.txnId}`);
      batch.set(txRef, {
        id: `${account.id}_${tx.txnId}`,
        accountId: account.id,
        amount: tx.type === 'DEBIT' ? -Math.round(tx.amount) : Math.round(tx.amount),
        category: 'uncategorized',
        merchant: tx.narration || 'Unknown',
        date: tx.valueDate,
        note: '',
        source: 'aa_sync',
      });
    }
  }

  await batch.commit();
  res.status(200).json({ success: true });
});
